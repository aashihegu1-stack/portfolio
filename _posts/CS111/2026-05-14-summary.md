(function() {

    const visibleCount = document.getElementById('visible-count');
    const noResults = document.getElementById('no-results');

    function updateCount() {
        const visible = Array.from(cards).filter(c => c.style.display !== 'none').length;
        visibleCount.textContent = visible;
        noResults.classList.toggle('show', visible === 0);
    }

    document.querySelectorAll('.filter-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            const filter = btn.dataset.filter;
            cards.forEach(c => {
                c.style.display = (filter === 'all' || c.dataset.category === filter) ? '' : 'none';
            });
            updateCount();
            btn.blur();
        });
    });

    document.getElementById('expand-all').addEventListener('click', () => {
        cards.forEach(c => { if (c.style.display !== 'none') c.open = true; });
    });
    document.getElementById('collapse-all').addEventListener('click', () => {
        cards.forEach(c => c.open = false);
    });
})();
